function submitData(){ let data={ months:document.getElementById("months").value, goal:document.getElementById("goal").value, income:[], food:[], transport:[], rent:[], fun:[], shopping:[] }; document.querySelectorAll(".income").forEach(e=>data.income.push(e.value)); document.querySelectorAll(".food").forEach(e=>data.food.push(e.value)); document.querySelectorAll(".transport").forEach(e=>data.transport.push(e.value)); document.querySelectorAll(".rent").forEach(e=>data.rent.push(e.value)); document.querySelectorAll(".fun").forEach(e=>data.fun.push(e.value)); document.querySelectorAll(".shopping").forEach(e=>data.shopping.push(e.value)); fetch("https://finance-qk18.onrender.com/calculate",{ method:"POST", headers:{"Content-Type":"application/json"}, body:JSON.stringify(data) }) .then(r=>r.json()) .then(res=>{ localStorage.setItem("financeData",JSON.stringify(res)); window.location.href="dashboard.html"; }); }
